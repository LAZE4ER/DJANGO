CREATE TABLE person (
    id INT PRIMARY KEY AUTO_INCREMENT,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    address VARCHAR(100),
    school_number INT
);
CREATE TABLE school (
    school_number INT PRIMARY KEY,
    school_address VARCHAR(100),
    school_name VARCHAR(100)
);
CREATE TABLE class (
    id INT PRIMARY KEY AUTO_INCREMENT,
    class_number VARCHAR(10),
    school_number INT,
    classroom VARCHAR(20)
);
